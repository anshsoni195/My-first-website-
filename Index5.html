const express = require('express'); const cors = require('cors'); const app = express(); const PORT = process.env.PORT || 5000; app.use(cors()); app.use(express.json()); // Mock Product Database (Can be expanded to MongoDB/MySQL later) const products = [ { id: 1, name: "Wireless Headphones", price: 99.99, image: "🎧" }, { id: 2, name: "Mechanical Keyboard", price: 129.99, image: "⌨️" }, { id: 3, name: "Gaming Mouse", price: 49.99, image: "🖱️" } ]; // 1. GET: Fetch product catalog app.get('/api/products', (req, res) => { res.json(products); }); // 2. POST: Handle Checkout / Place Order app.post('/api/orders', (req, res) => { const { cartItems, total } = req.body; if (!cartItems || cartItems.length === 0) { return res.status(400).json({ error: "Your cart is empty." }); } // Simulate order tracking creation const orderId = "ORD-" + Math.floor(100000 + Math.random() * 900000); res.status(201).json({ success: true, orderId: orderId, message: "Order placed successfully! Tracking active.", total: total }); }); app.listen(PORT, () => console.log(`E-commerce API running on port ${PORT}`));
